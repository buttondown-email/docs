# Tags

## The tag object

A tag looks like this:

<ObjectDescription
  example={JSON.stringify(DATA.object, null, 4)}
  fields={[
    { field: "creation_date", type: "datetime", description: "" },
    { field: "id", type: "uuid" },
    { field: "name", type: "string" },
    { field: "description", type: "string" },
    { field: "color", type: "string" },
  ]}
/>

<Endpoint title="Listing tags" method="GET" path={DATA.endpoint} />
<ParametersTable
  content={[
    {
      parameter: "page",
      type: "string",
      description: "A page number within the paginated result set.",
      example: "",
      optional: true,
    },
  ]}
/>
<ResponsesTable
  content={[
    {
      Status: "200",
      description: "",
      "Sample Response": {
        count: 1,
        next: "foo",
        previous: null,
        results: [DATA.object],
      },
    },
  ]}
/>

---

<Endpoint title="Creating a new tag" method="POST" path={DATA.endpoint} />
<ParametersTable
  content={[
    {
      parameter: "name",
      type: "string",
      description: "",
      optional: false,
    },
    {
      parameter: "color",
      type: "string",
      description: "",
      optional: true,
    },
    {
      parameter: "description",
      type: "string",
      description: "",
      optional: true,
    },
  ]}
/>
<ResponsesTable
  content={[
    {
      Status: "201",
      description: "",
      "Sample Response": DATA.object,
    },
  ]}
/>

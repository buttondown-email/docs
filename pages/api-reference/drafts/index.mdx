# Drafts

## The draft object

A draft looks like this:

<ObjectDescription
  example={JSON.stringify(DATA.object, null, 4)}
  fields={[
    { field: "creation_date", type: "datetime", description: "" },
    { field: "id", type: "uuid" },
    { field: "subject", type: "string" },
    { field: "body", type: "string" },
  ]}
/>

<Endpoint title="Listing drafts" method="GET" path={DATA.endpoint} />
<ParametersTable
  content={[
    {
      parameter: "page",
      type: "string",
      description: "A page number within the paginated result set.",
      example: "",
      optional: true,
    },
  ]}
/>
<ResponsesTable
  content={[
    {
      Status: "200",
      description: "",
      "Sample Response": {
        count: 1,
        next: "foo",
        previous: null,
        results: [DATA.object],
      },
    },
  ]}
/>

<Endpoint title="Creating a new draft" method="POST" path={DATA.endpoint} />
